var tween =	TweenMax.to(".box19", 10, {left:"90%", ease: Expo.easeOut});
$(".btn19_1").click(function(){tween.play()});

$(".btn19_3").click(function(){tween.resume()});
$(".btn19_4").click(function(){tween.reverse()});
$(".btn19_5").click(function(){tween.seek(0.5)});
$(".btn19_6").click(function(){tween.timeScale(9.5)});
$(".btn19_7").click(function(){tween.timeScale(2)});
$(".btn19_8").click(function(){tween.kill()});
$(".btn19_9").click(function(){tween.restart()});
